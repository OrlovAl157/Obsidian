## 📦 Dataclass с типами

Dataclass’ы отлично работают с типизацией:

```python
from dataclasses import dataclass, field
from typing import Optional
from datetime import datetime

@dataclass
class User:
    """Класс пользователя"""
    id: int
    name: str
    email: str
    created_at: datetime = field(default_factory=datetime.now)
    bio: Optional[str] = None
    tags: list[str] = field(default_factory=list)
    
    def is_adult(self) -> bool:
        return len(self.name) > 2
    
    def add_tag(self, tag: str) -> None:
        self.tags.append(tag)

# Использование
user = User(
    id=1,
    name="Alice",
    email="alice@example.com",
    bio="Developer"
)

print(user.is_adult())
user.add_tag("python")
```

## 🌐 API клиент

Типизация для работы с API:

```python
from dataclasses import dataclass
from typing import TypedDict, Generic, TypeVar
import json

class UserResponse(TypedDict):
    """API ответ для пользователя"""
    id: int
    name: str
    email: str

@dataclass
class ApiResponse(Generic[T]):
    """Обёртка для API ответа"""
    status: int
    data: T
    error: str | None = None

class ApiClient:
    def __init__(self, base_url: str):
        self.base_url = base_url
    
    def get_user(self, user_id: int) -> UserResponse:
        """Получить пользователя"""
        # В реальности: requests.get(...)
        return UserResponse(
            id=user_id,
            name="Alice",
            email="alice@example.com"
        )
    
    def list_users(self) -> list[UserResponse]:
        """Список пользователей"""
        return [
            UserResponse(id=1, name="Alice", email="alice@example.com"),
            UserResponse(id=2, name="Bob", email="bob@example.com")
        ]

# Использование
client = ApiClient("https://api.example.com")
user = client.get_user(1)
print(user["name"])  # ✅ mypy знает тип

users = client.list_users()
for u in users:
    print(u["email"])  # ✅ mypy знает тип
```

## 🗄️ Работа с БД

Generic репозиторий для работы с БД:

```python
from dataclasses import dataclass
from typing import Generic, TypeVar, Protocol

T = TypeVar('T')
ID = TypeVar('ID')

@dataclass
class BaseModel:
    id: int

@dataclass
class User(BaseModel):
    name: str
    email: str

@dataclass
class Product(BaseModel):
    title: str
    price: float

class Repository(Generic[T, ID]):
    """Generic репозиторий"""
    
    def __init__(self):
        self.items: dict[ID, T] = {}  # type: ignore
    
    def save(self, id: ID, item: T) -> None:
        self.items[id] = item
    
    def find(self, id: ID) -> T | None:
        return self.items.get(id)
    
    def find_all(self) -> list[T]:
        return list(self.items.values())
    
    def delete(self, id: ID) -> bool:
        if id in self.items:
            del self.items[id]
            return True
        return False

# Использование
user_repo: Repository[User, int] = Repository()
product_repo: Repository[Product, int] = Repository()

user_repo.save(1, User(id=1, name="Alice", email="alice@example.com"))
product_repo.save(1, Product(id=1, title="Laptop", price=999.99))

user = user_repo.find(1)
if user:
    print(f"User: {user.name}")
```

## ⚙️ Конфигурация

Типизация конфигурации:

```python
from dataclasses import dataclass
from typing import Literal
from enum import Enum

class Environment(str, Enum):
    DEVELOPMENT = "development"
    PRODUCTION = "production"
    TESTING = "testing"

@dataclass
class DatabaseConfig:
    """Конфиг БД"""
    host: str
    port: int
    username: str
    password: str
    database: str

@dataclass
class AppConfig:
    """Конфиг приложения"""
    env: Environment
    debug: bool
    database: DatabaseConfig
    log_level: Literal['DEBUG', 'INFO', 'WARNING', 'ERROR']
    max_workers: int = 4

# Использование
config = AppConfig(
    env=Environment.PRODUCTION,
    debug=False,
    database=DatabaseConfig(
        host="localhost",
        port=5432,
        username="user",
        password="pass",
        database="mydb"
    ),
    log_level='INFO',
    max_workers=8
)

print(f"Running in {config.env.value}")
print(f"Database: {config.database.host}:{config.database.port}")
```

## 💻 Полные примеры

### Пример 1: Полный Service layer

```python
from dataclasses import dataclass
from typing import Generic, TypeVar
from datetime import datetime

@dataclass
class User:
    id: int
    name: str
    email: str
    created_at: datetime

class UserRepository:
    """Слой доступа к БД"""
    
    def __init__(self):
        self.users: dict[int, User] = {}
    
    def save(self, user: User) -> None:
        self.users[user.id] = user
    
    def find_by_id(self, user_id: int) -> User | None:
        return self.users.get(user_id)
    
    def find_by_email(self, email: str) -> User | None:
        for user in self.users.values():
            if user.email == email:
                return user
        return None

class UserService:
    """Бизнес-логика"""
    
    def __init__(self, repo: UserRepository):
        self.repo = repo
    
    def create_user(self, name: str, email: str) -> User:
        """Создать пользователя"""
        user_id = len(self.repo.users) + 1
        user = User(
            id=user_id,
            name=name,
            email=email,
            created_at=datetime.now()
        )
        self.repo.save(user)
        return user
    
    def get_user(self, user_id: int) -> User | None:
        return self.repo.find_by_id(user_id)
    
    def find_user_by_email(self, email: str) -> User | None:
        return self.repo.find_by_email(email)

# Использование
repo = UserRepository()
service = UserService(repo)

user = service.create_user("Alice", "alice@example.com")
print(f"Created: {user.name}")

found = service.get_user(1)
if found:
    print(f"Found: {found.email}")
```

### Пример 2: Асинхронный код

```python
import asyncio
from typing import Callable, Awaitable

async def fetch_user(user_id: int) -> dict[str, str]:
    """Получить пользователя асинхронно"""
    await asyncio.sleep(0.1)
    return {"id": str(user_id), "name": "Alice"}

async def fetch_posts(user_id: int) -> list[dict[str, str]]:
    """Получить посты асинхронно"""
    await asyncio.sleep(0.1)
    return [
        {"id": "1", "title": "Post 1"},
        {"id": "2", "title": "Post 2"}
    ]

async def main() -> None:
    """Основная функция"""
    user = await fetch_user(1)
    posts = await fetch_posts(1)
    print(f"User: {user['name']}")
    print(f"Posts: {len(posts)}")

# Использование
asyncio.run(main())
```

### Пример 3: Декоратор с сохранением типов

```python
from typing import ParamSpec, TypeVar, Callable
from functools import wraps
import time

P = ParamSpec('P')
R = TypeVar('R')

def timeit(func: Callable[P, R]) -> Callable[P, R]:
    """Декоратор для измерения времени"""
    @wraps(func)
    def wrapper(*args: P.args, **kwargs: P.kwargs) -> R:
        start = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"{func.__name__} took {elapsed:.4f}s")
        return result
    return wrapper  # type: ignore

@timeit
def slow_add(a: int, b: int) -> int:
    time.sleep(0.1)
    return a + b

result = slow_add(5, 3)  # Сигнатура сохранена!
print(f"Result: {result}")
```

## ✅ Главные правила для production

✅ **Всегда типизируйте функции** — особенно публичные API  
✅ **Используйте Dataclass** — вместо обычных классов для моделей  
✅ **Generic’и для переиспользования** — репозитории, сервисы  
✅ **TypedDict для API** — когда работаешь со словарями  
✅ **Protocol для интерфейсов** — чистая архитектура  
✅ **Запускайте mypy в CI/CD** — автоматизируйте проверку  
✅ **Документируйте сложные типы** — комментарии помогают
