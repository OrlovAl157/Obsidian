# 👥 Типизация методов и классов

> **self** не нужно типизировать явно, но методы типизируются как обычные функции. Когда метод возвращает сам класс (напр., для chaining), используется имя класса в кавычках или `typing.Self`. Типизация `__init__` — важная часть, так как это первый контакт с классом.

## 🔧 Базовая типизация методов

**self не типизируется**, типизируются параметры и возврат:

```python
class Calculator:
    def add(self, a: int, b: int) -> int:
        """Сложение двух чисел"""
        return a + b
    
    def divide(self, dividend: float, divisor: float) -> float:
        """Деление"""
        if divisor == 0:
            raise ValueError("Cannot divide by zero")
        return dividend / divisor
    
    def get_info(self) -> str:
        """Получить информацию"""
        return "I am a calculator"

# Использование
calc = Calculator()
result = calc.add(5, 3)  # 8 (int)
quotient = calc.divide(10.0, 2.0)  # 5.0 (float)
info = calc.get_info()  # "I am a calculator" (str)
```

**Важно:** `self` всегда первый параметр, но не указывается в типах!

## 📝 Типизация **init**

`__init__` всегда возвращает `None`:

```python
class Person:
    """Класс человека"""
    
    def __init__(self, name: str, age: int) -> None:
        """Инициализация"""
        self.name = name
        self.age = age
    
    def birthday(self) -> None:
        """День рождения"""
        self.age += 1

class User:
    """Пользователь с опциональным email"""
    
    def __init__(
        self,
        username: str,
        email: str | None = None,
        is_active: bool = True
    ) -> None:
        self.username = username
        self.email = email
        self.is_active = is_active

# Использование
person = Person("Alice", 30)
person.birthday()

user = User("alice", "alice@example.com")
```

## 🔄 Методы, возвращающие self

Когда метод возвращает сам объект (для method chaining):

```python
class StringBuilder:
    """Builder для строк"""
    
    def __init__(self) -> None:
        self.text = ""
    
    def add(self, text: str) -> "StringBuilder":
        """Добавить текст"""
        self.text += text
        return self
    
    def add_line(self, text: str) -> "StringBuilder":
        """Добавить строку"""
        self.text += text + "\n"
        return self
    
    def build(self) -> str:
        """Получить результат"""
        return self.text

# Использование (chaining)
result = (StringBuilder()
    .add("Hello")
    .add(" ")
    .add("World")
    .add_line("!")
    .build())

print(result)  # "Hello World!\n"
```

**Python 3.11+** — можно использовать `Self` из `typing`:

```python
from typing import Self

class StringBuilder:
    def __init__(self) -> None:
        self.text = ""
    
    def add(self, text: str) -> Self:  # Вместо "StringBuilder"
        self.text += text
        return self
```

## 🏗️ Class методы и static методы

**@classmethod** — получает класс вместо self:

```python
class User:
    """Пользователь"""
    count = 0
    
    def __init__(self, name: str) -> None:
        self.name = name
        User.count += 1
    
    @classmethod
    def from_string(cls, data: str) -> "User":
        """Создать из строки (паттерн Factory)"""
        name = data.split(",")[0]
        return cls(name)
    
    @classmethod
    def get_count(cls) -> int:
        """Получить количество пользователей"""
        return cls.count

# Использование
user1 = User("Alice")
user2 = User.from_string("Bob,bob@example.com")
print(User.get_count())  # 2
```

**@staticmethod** — нет self и cls:

```python
class MathUtils:
    """Вспомогательные математические функции"""
    
    @staticmethod
    def is_even(n: int) -> bool:
        """Проверить, чётное ли число"""
        return n % 2 == 0
    
    @staticmethod
    def factorial(n: int) -> int:
        """Факториал"""
        if n <= 1:
            return 1
        return n * MathUtils.factorial(n - 1)

# Использование
print(MathUtils.is_even(4))  # True
print(MathUtils.factorial(5))  # 120
```

## 💻 Примеры

### Пример 1: Класс с типизацией

```python
from typing import Optional

class BankAccount:
    """Банковский счёт"""
    
    def __init__(self, owner: str, balance: float = 0.0) -> None:
        """Создать счёт"""
        self.owner = owner
        self.balance = balance
    
    def deposit(self, amount: float) -> None:
        """Пополнить счёт"""
        if amount <= 0:
            raise ValueError("Amount must be positive")
        self.balance += amount
    
    def withdraw(self, amount: float) -> bool:
        """Снять деньги (вернёт True если успешно)"""
        if amount <= 0:
            raise ValueError("Amount must be positive")
        if amount > self.balance:
            return False
        self.balance -= amount
        return True
    
    def get_balance(self) -> float:
        """Получить баланс"""
        return self.balance
    
    def get_info(self) -> str:
        """Получить информацию"""
        return f"{self.owner}: ${self.balance:.2f}"

# Использование
account = BankAccount("Alice", 100.0)
account.deposit(50.0)
success = account.withdraw(30.0)
print(account.get_info())  # "Alice: $120.00"
```

### Пример 2: Builder с chaining

```python
class QueryBuilder:
    """Builder для SQL запросов"""
    
    def __init__(self) -> None:
        self.select_clause = ""
        self.from_clause = ""
        self.where_clause = ""
        self.limit_value: int | None = None
    
    def select(self, columns: str) -> "QueryBuilder":
        """SELECT"""
        self.select_clause = f"SELECT {columns}"
        return self
    
    def from_table(self, table: str) -> "QueryBuilder":
        """FROM"""
        self.from_clause = f"FROM {table}"
        return self
    
    def where(self, condition: str) -> "QueryBuilder":
        """WHERE"""
        self.where_clause = f"WHERE {condition}"
        return self
    
    def limit(self, count: int) -> "QueryBuilder":
        """LIMIT"""
        self.limit_value = count
        return self
    
    def build(self) -> str:
        """Построить запрос"""
        query = self.select_clause
        if self.from_clause:
            query += f" {self.from_clause}"
        if self.where_clause:
            query += f" {self.where_clause}"
        if self.limit_value:
            query += f" LIMIT {self.limit_value}"
        return query

# Использование
query = (QueryBuilder()
    .select("id, name, email")
    .from_table("users")
    .where("age > 18")
    .limit(10)
    .build())

print(query)
# SELECT id, name, email FROM users WHERE age > 18 LIMIT 10
```

### Пример 3: Атрибуты класса

```python
from typing import ClassVar

class Dog:
    """Класс собаки"""
    
    # Атрибут класса (общий для всех экземпляров)
    species: ClassVar[str] = "Canis familiaris"
    count: ClassVar[int] = 0
    
    # Атрибуты экземпляра
    name: str
    age: int
    
    def __init__(self, name: str, age: int) -> None:
        self.name = name
        self.age = age
        Dog.count += 1
    
    def describe(self) -> str:
        """Описать собаку"""
        return f"{self.name} is a {Dog.species}, {self.age} years old"
    
    @classmethod
    def get_total_dogs(cls) -> int:
        """Получить всех собак"""
        return cls.count

# Использование
dog1 = Dog("Rex", 5)
dog2 = Dog("Bella", 3)

print(dog1.describe())  # "Rex is a Canis familiaris, 5 years old"
print(Dog.get_total_dogs())  # 2
```

## ⚠️ Частые ошибки

```python
# ❌ Типизировали self (не нужно!)
class MyClass:
    def method(self: "MyClass", x: int) -> None:  # Лишнее!
        pass

# ✅ Правильно (self не типизируем)
class MyClass:
    def method(self, x: int) -> None:
        pass

# ❌ Забыли -> None для __init__
class Person:
    def __init__(self, name: str):  # Нет return type!
        self.name = name

# ✅ Правильно
class Person:
    def __init__(self, name: str) -> None:
        self.name = name

# ❌ Неправильный return type для self
class Builder:
    def add(self, value: str) -> "Builder":  # OK, но есть лучше
        return self

# ✅ Python 3.11+
from typing import Self

class Builder:
    def add(self, value: str) -> Self:  # Лучше!
        return self

# ❌ Забыли -> None для классметода без return
class Config:
    @classmethod
    def setup(cls):  # Нет типа!
        pass

# ✅ Правильно
class Config:
    @classmethod
    def setup(cls) -> None:
        pass

# ❌ cls типизировать как обычный параметр
class Factory:
    @classmethod
    def create(cls: type["Factory"]) -> "Factory":  # Лишнее!
        return cls()

# ✅ Правильно (cls автоматический)
class Factory:
    @classmethod
    def create(cls) -> "Factory":
        return cls()
```

## ✅ Главные правила

✅ **self не типизируется** — это неявно  
✅ **Методы типизируются как функции** — параметры и return  
✅ ****init** всегда -> None** — никогда не возвращает значение  
✅ **Возврат self в кавычках** — `-> "ClassName"` или `-> Self`  
✅ **@classmethod получает cls** — первый параметр  
✅ **@staticmethod без self и cls** — как обычная функция  
✅ **ClassVar для атрибутов класса** — отличает от атрибутов экземпляра
