# 🎯 Практические примеры типизации

## 📌 Коротко

> Реальные примеры использования типизации в практических сценариях — от простых функций до сложных классов с generic типами.

---

## 💡 Полезные советы

- **Начните с простого** — типизируйте сигнатуры функций
- **Постепенно добавляйте типы** — не переусложняйте сразу
- **Используйте type checker** — mypy будет вам помогать
- **Документируйте сложные типы** — комментарии помогают
- **Тестируйте типы** — напишите тесты для типизированного кода

## 📖 Примеры

### Пример 1: API функции

```python
from typing import Optional, List, Dict

def get_user(user_id: int) -> Optional[Dict[str, str]]:
    """Получить пользователя по ID"""
    users = {
        1: {"name": "Alice", "email": "alice@example.com"},
        2: {"name": "Bob", "email": "bob@example.com"}
    }
    return users.get(user_id)

def get_all_users() -> List[Dict[str, str]]:
    """Получить всех пользователей"""
    return [
        {"name": "Alice", "email": "alice@example.com"},
        {"name": "Bob", "email": "bob@example.com"}
    ]

# Использование
user = get_user(1)
if user:
    print(f"User: {user['name']}")

all_users = get_all_users()
for user in all_users:
    print(user["name"])
```

### Пример 2: Обработка данных

```python
from typing import Callable, List, TypeVar, Generic

T = TypeVar('T')
U = TypeVar('U')

def map_data(items: List[T], transform: Callable[[T], U]) -> List[U]:
    """Преобразовать список"""
    return [transform(item) for item in items]

def filter_data(items: List[T], predicate: Callable[[T], bool]) -> List[T]:
    """Отфильтровать список"""
    return [item for item in items if predicate(item)]

# Использование
numbers = [1, 2, 3, 4, 5]
strings = map_data(numbers, str)        # List[str]
evens = filter_data(numbers, lambda x: x % 2 == 0)  # List[int]
```

### Пример 3: Класс с типизацией

```python
from typing import Generic, TypeVar, List, Optional

T = TypeVar('T')

class Repository(Generic[T]):
    """Репозиторий для хранения данных"""
    
    def __init__(self) -> None:
        self._items: List[T] = []
    
    def add(self, item: T) -> None:
        self._items.append(item)
    
    def get(self, index: int) -> Optional[T]:
        if 0 <= index < len(self._items):
            return self._items[index]
        return None
    
    def get_all(self) -> List[T]:
        return self._items.copy()

# Использование
string_repo: Repository[str] = Repository()
string_repo.add("Hello")
string_repo.add("World")
all_strings = string_repo.get_all()  # List[str]
```

### Пример 4: Обработка ошибок

```python
from typing import Union, Tuple

def divide(a: float, b: float) -> Union[float, str]:
    """Деление с обработкой ошибок"""
    if b == 0:
        return "Error: Division by zero"
    return a / b

def safe_divide(a: float, b: float) -> Tuple[bool, Union[float, str]]:
    """Деление с индикатором успеха"""
    if b == 0:
        return False, "Error: Division by zero"
    return True, a / b

# Использование
result = divide(10, 2)
if isinstance(result, str):
    print(f"Error: {result}")
else:
    print(f"Result: {result}")

success, value = safe_divide(10, 2)
if success:
    print(f"Result: {value}")
else:
    print(f"Error: {value}")
```

### Пример 5: Config класс

```python
from typing import Optional, Dict, Any
from dataclasses import dataclass

@dataclass
class DatabaseConfig:
    """Конфигурация базы данных"""
    host: str
    port: int
    username: str
    password: str
    database: str
    ssl: bool = False
    timeout: int = 30
    
    def get_connection_string(self) -> str:
        """Собрать строку подключения"""
        protocol = "postgresql+psycopg2" if self.ssl else "postgresql"
        return (f"{protocol}://{self.username}:{self.password}"
                f"@{self.host}:{self.port}/{self.database}")

# Использование
config = DatabaseConfig(
    host="localhost",
    port=5432,
    username="admin",
    password="secret",
    database="myapp"
)
connection_string = config.get_connection_string()
```

### Пример 6: Декоратор с типами

```python
from typing import Callable, TypeVar, Any
from functools import wraps
import time

F = TypeVar('F', bound=Callable[..., Any])

def timer(func: F) -> F:
    """Декоратор для измерения времени выполнения"""
    @wraps(func)
    def wrapper(*args: Any, **kwargs: Any) -> Any:
        start = time.time()
        result = func(*args, **kwargs)
        duration = time.time() - start
        print(f"{func.__name__} took {duration:.4f}s")
        return result
    return wrapper  # type: ignore

@timer
def slow_function(n: int) -> int:
    """Медленная функция"""
    time.sleep(0.1)
    return n * 2

result = slow_function(42)  # "slow_function took 0.1XXXs"
```

### Пример 7: Async функции

```python
from typing import Optional, List
import asyncio

async def fetch_user(user_id: int) -> Optional[dict]:
    """Асинхронно получить пользователя"""
    await asyncio.sleep(0.1)  # Имитация сетевого запроса
    return {"id": user_id, "name": f"User {user_id}"}

async def fetch_all_users(user_ids: List[int]) -> List[dict]:
    """Асинхронно получить всех пользователей"""
    tasks = [fetch_user(uid) for uid in user_ids]
    results = await asyncio.gather(*tasks)
    return [u for u in results if u is not None]

# Использование
async def main() -> None:
    users = await fetch_all_users([1, 2, 3])
    for user in users:
        print(f"User: {user['name']}")

asyncio.run(main())
```

## 🎯 Best Practices

### Структурированный подход

```python
# 1. Импорты с типами
from typing import List, Optional, Dict, Callable, TypeVar

# 2. Type variables
T = TypeVar('T')

# 3. Constants и типы
MAX_RETRIES: int = 3
TIMEOUT: float = 30.0

# 4. Функции с типами
def process(data: List[T]) -> Optional[T]:
    """Обработать данные"""
    return data[0] if data else None

# 5. Классы с типами
class Service:
    def __init__(self) -> None:
        self._cache: Dict[str, str] = {}
    
    def get(self, key: str) -> Optional[str]:
        return self._cache.get(key)
```
