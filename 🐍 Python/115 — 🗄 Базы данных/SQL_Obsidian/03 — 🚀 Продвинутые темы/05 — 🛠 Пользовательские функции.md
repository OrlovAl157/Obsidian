# 🧩 Пользовательские функции (UDF)

## Синтаксис создания

```sql
DELIMITER //
CREATE FUNCTION function_name(param1 TYPE, param2 TYPE, ...)
RETURNS return_type
{DETERMINISTIC | NOT DETERMINISTIC}
{READS SQL DATA | NO SQL}
COMMENT 'здесь можно оставить комментарий для чего нужна функция'
BEGIN
    -- тело функции
    RETURN return_value;
END //
DELIMITER ;
```

## Виды функций

| Ключевое слово | Значение |
|----------------|----------|
| `DETERMINISTIC` | возвращает один и тот же результат при одинаковых входных данных |
| `NOT DETERMINISTIC` | возвращает разные результаты, например с `RAND()` |
| `READS SQL DATA` | работает с данными из базы данных |
| `NO SQL` | не обращается к таблицам базы данных |

## Переменные внутри функции

```sql
DECLARE variable_name data_type;
```

## Удаление функции

```sql
DROP FUNCTION function_name;
```

## Пример 1: чётное / нечётное число

```sql
DELIMITER //
CREATE FUNCTION IS_EVEN(number INT)
RETURNS INT
DETERMINISTIC
BEGIN
    RETURN IF(number MOD 2 = 0, 1, 0);
END //
DELIMITER ;

SELECT IS_EVEN(10), IS_EVEN(15);
-- Результат: 1 | 0
```

## Пример 2: с подзапросом

```sql
DELIMITER //
CREATE FUNCTION STUDENTS_WITH_GRADE(score INT)
RETURNS INT
NOT DETERMINISTIC
READS SQL DATA
BEGIN
    RETURN (SELECT COUNT(*)
            FROM Math
            WHERE grade = score);
END //
DELIMITER ;

SELECT STUDENTS_WITH_GRADE(4), STUDENTS_WITH_GRADE(5);
-- Результат: 2 | 3
```

## Пример 3: с переменными внутри функции

```sql
DELIMITER //
CREATE FUNCTION MIN_MAX_GRADE()
RETURNS TEXT
NOT DETERMINISTIC
READS SQL DATA
BEGIN
    DECLARE min_grade INT;
    DECLARE max_grade INT;

    SET min_grade := (SELECT MIN(grade) FROM Math);
    SET max_grade := (SELECT MAX(grade) FROM Math);

    RETURN CONCAT('Lowest grade: ', min_grade, ', Highest grade: ', max_grade);
END //
DELIMITER ;

SELECT MIN_MAX_GRADE();
-- Результат: Lowest grade: 1, Highest grade: 5
```

## Пример 4: с IF-ELSEIF-ELSE

```sql
DELIMITER //
CREATE FUNCTION IS_EVEN(number INT)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE result INT;

    IF number MOD 2 = 0 THEN
        SET result := 1;
    ELSEIF number MOD 2 IS NULL THEN
        SET result := NULL;
    ELSE
        SET result := 0;
    END IF;

    RETURN result;
END //
DELIMITER ;

SELECT IS_EVEN(10), IS_EVEN(15);
-- Результат: 1 | 0
```

## Пример 5: с WHILE

```sql
DELIMITER //
CREATE FUNCTION SUM_UP_TO_N(n INT)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE result INT DEFAULT 0;
    DECLARE count INT DEFAULT 1;

    WHILE count <= n DO
        SET result := result + count;
        SET count := count + 1;
    END WHILE;

    RETURN result;
END //
DELIMITER ;

SELECT SUM_UP_TO_N(1),   -- 1
       SUM_UP_TO_N(2),   -- 1 + 2 = 3
       SUM_UP_TO_N(3),   -- 1 + 2 + 3 = 6
       SUM_UP_TO_N(4),   -- 1 + 2 + 3 + 4 = 10
       SUM_UP_TO_N(5);   -- 1 + 2 + 3 + 4 + 5 = 15
```
