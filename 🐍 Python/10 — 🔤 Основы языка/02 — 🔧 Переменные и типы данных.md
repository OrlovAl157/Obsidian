# 🔧 Переменные и типы данных

## 📌 Коротко

> Переменная — контейнер для хранения значений. Python — динамически типизированный язык, поэтому тип определяется автоматически при присваивании.

---

## 💡 Полезные советы

- **Имена переменных в snake_case** — `my_variable` а не `myVariable`
- **Используйте осмысленные имена** — `user_age` вместо `a` или `temp`
- **Переменные чувствительны к регистру** — `Age` и `age` это разные переменные
- **Тип можно узнать через type()** — `print(type(x))`
- **Переменные могут менять тип** — `x = 5; x = "text"` допустимо, но не рекомендуется

---

## 🚀 Быстрая справка

| Тип | Описание | Примеры |
|---|---|---|
| **int** | Целое число | `42`, `-10`, `0` |
| **float** | Дробное число | `3.14`, `-2.5`, `1.0` |
| **str** | Текст/строка | `"Hello"`, `'World'` |
| **bool** | Логическое значение | `True`, `False` |
| **None** | Отсутствие значения | `None` |
| **list** | Список (изменяемый) | `[1, 2, 3]` |
| **dict** | Словарь (пары ключ-значение) | `{"key": "value"}` |
| **tuple** | Кортеж (неизменяемый) | `(1, 2, 3)` |
| **set** | Множество (уникальные) | `{1, 2, 3}` |

## 📖 Полные примеры

### Создание переменных - числа

```python
# Целые числа
age = 25
temperature = -5
year = 2024
zero = 0
big_number = 1000000

print(age)          # 25
print(type(age))    # <class 'int'>

# Дробные числа (float)
height = 1.75
pi = 3.14159
weight = 65.5
percentage = 99.9

print(height)       # 1.75
print(type(height)) # <class 'float'>

# Большие числа (подчёркивания для читаемости)
million = 1_000_000
billion = 1_000_000_000
print(million)      # 1000000
```

### Создание переменных - строки

```python
# Строки в двойных кавычках
name = "Alice"
message = "Hello, World!"
empty = ""

# Строки в одинарных кавычках
greeting = 'Hi'
path = 'C:\\Users\\Alice\\file.txt'

# Тройные кавычки (многострочные)
text = """
This is
a multiline
string
"""

# Специальные символы
tab = "Hello\tWorld"   # \t = табуляция
newline = "Hello\nWorld"  # \n = новая строка
quote = 'He said "Hi"'  # Кавычки в одинарных

print(name)         # Alice
print(type(name))   # <class 'str'>
```

### Создание переменных - логические и None

```python
# Логические значения (bool)
is_active = True
is_admin = False
has_permission = True

print(is_active)    # True
print(type(is_active))  # <class 'bool'>

# None - отсутствие значения
value = None
result = None

print(value)        # None
print(type(value))  # <class 'NoneType'>
```

### Определение типа переменной

```python
x = 42
print(type(x))      # <class 'int'>

y = 3.14
print(type(y))      # <class 'float'>

name = "Bob"
print(type(name))   # <class 'str'>

flag = True
print(type(flag))   # <class 'bool'>

nothing = None
print(type(nothing))  # <class 'NoneType'>

items = [1, 2, 3]
print(type(items))  # <class 'list'>
```

### Преобразование типов (Type Casting)

```python
# Строка в число
text = "42"
number = int(text)
print(number)       # 42
print(type(number)) # <class 'int'>

# Число в строку
num = 100
text = str(num)
print(text)         # "100"
print(type(text))   # <class 'str'>

# Дробное в целое
value = 3.7
whole = int(value)
print(whole)        # 3 (отбрасывает дробную часть)

# Строка в дробное
text = "3.14"
decimal = float(text)
print(decimal)      # 3.14

# Целое в дробное
x = 5
y = float(x)
print(y)            # 5.0

# В логическое значение
print(bool(1))      # True
print(bool(0))      # False
print(bool("text")) # True
print(bool(""))     # False
print(bool([]))     # False
print(bool([1]))    # True
```

### Множественное присваивание

```python
# Одно значение нескольким переменным
x = y = z = 0
print(x, y, z)      # 0 0 0

# Распаковка кортежа
a, b, c = 1, 2, 3
print(a, b, c)      # 1 2 3

# Распаковка списка
values = [10, 20, 30]
x, y, z = values
print(x, y, z)      # 10 20 30

# Обмен значений (Python magic!)
x = 10
y = 20
x, y = y, x
print(x, y)         # 20 10

# С пропуском переменной
a, _, c = 1, 2, 3
print(a, c)         # 1 3
```

### Константы (по соглашению)

```python
# Python соглашение - CAPS для констант
# Они не защищены, но сигнализируют что менять нельзя

PI = 3.14159
MAX_SIZE = 100
TIMEOUT = 30
DEBUG = True

# Используем константы
radius = 5
area = PI * radius ** 2
print(area)         # 78.53975

# Менять не рекомендуется (но Python позволит)
MAX_SIZE = 200      # ⚠️ Плохая идея!
```

### Переименование переменных

```python
# Присваивание значения другой переменной
x = 5
y = x
print(y)            # 5

# Изменение одной не влияет на другую (для примитивных типов)
x = 10
print(y)            # 5 (не изменилось)

# Для списков - ссылка на один объект
list1 = [1, 2, 3]
list2 = list1
list1.append(4)
print(list2)        # [1, 2, 3, 4] (изменилось!)

# Чтобы скопировать список
list3 = list1.copy()
list1.append(5)
print(list3)        # [1, 2, 3, 4] (не изменилось)
```

## 🎯 Практические примеры

### Персональная информация

```python
# Данные о человеке
name = "Alice Johnson"
age = 28
height = 1.75
email = "alice@example.com"
is_student = False

# Вывод информации
print(f"Name: {name}")
print(f"Age: {age} years")
print(f"Height: {height}m")
print(f"Email: {email}")
print(f"Student: {is_student}")

# Расчёты
birth_year = 2024 - age
print(f"Born in: {birth_year}")
```

### Калькулятор расходов

```python
# Цена товаров
item1_price = 10.99
item1_qty = 2

item2_price = 5.50
item2_qty = 3

# Расчёты
total1 = item1_price * item1_qty
total2 = item2_price * item2_qty
subtotal = total1 + total2

tax_rate = 0.08
tax = subtotal * tax_rate
total = subtotal + tax

# Вывод
print(f"Item 1: ${item1_price} x {item1_qty} = ${total1:.2f}")
print(f"Item 2: ${item2_price} x {item2_qty} = ${total2:.2f}")
print(f"Subtotal: ${subtotal:.2f}")
print(f"Tax ({tax_rate*100}%): ${tax:.2f}")
print(f"Total: ${total:.2f}")
```

### Химический расчёт (молекулярный вес)

```python
# Атомные веса
carbon = 12.01
hydrogen = 1.008
oxygen = 16.00

# Молекула: C6H12O6 (глюкоза)
c_atoms = 6
h_atoms = 12
o_atoms = 6

# Расчёт молекулярного веса
molecular_weight = (c_atoms * carbon) + \
                  (h_atoms * hydrogen) + \
                  (o_atoms * oxygen)

print(f"Glucose (C6H12O6)")
print(f"Molecular weight: {molecular_weight:.2f} g/mol")
```

### Конвертирование единиц

```python
# Конвертирование температуры Цельсия в Фаренгейт
celsius = 25
fahrenheit = (celsius * 9/5) + 32
print(f"{celsius}°C = {fahrenheit}°F")

# Конвертирование км в мили
kilometers = 100
miles = kilometers * 0.621371
print(f"{kilometers} km = {miles:.2f} miles")

# Конвертирование килограммов в фунты
kilograms = 75
pounds = kilograms * 2.20462
print(f"{kilograms} kg = {pounds:.2f} lbs")
```

## 📊 Типы данных и их особенности

### Числовые типы

```python
# int - целые числа (любой размер)
x = 42
x = -100
x = 0
x = 999_999_999_999

# float - дробные числа (приблизительные)
y = 3.14
y = -2.5
y = 1.0
y = 1e10  # 10000000000.0 (научная нотация)

# Арифметика со смешанными типами
result = 5 + 2.5     # 7.5 (результат float)
result = 10 / 3      # 3.333... (деление = float)
result = 10 // 3     # 3 (целое деление = int)
```

### Строковые типы

```python
# Одинарные и двойные кавычки эквивалентны
str1 = 'Hello'
str2 = "Hello"
print(str1 == str2)  # True

# Специальные символы (escape sequences)
escaped = "Line 1\nLine 2"      # \n = новая строка
tab = "Column1\tColumn2"        # \t = табуляция
backslash = "C:\\Users\\file"   # \\ = обратный слэш
quote = "He said \"Hi\""        # \" = кавычка

# Raw strings (буквально, без интерпретации)
path = r"C:\Users\file"         # Буквально: C:\Users\file

# Длина строки
text = "Hello"
print(len(text))                # 5
```

### Логические операции с типами

```python
# Все эти значения считаются False
print(bool(0))           # False
print(bool(0.0))         # False
print(bool(""))          # False
print(bool([]))          # False
print(bool({}))          # False
print(bool(None))        # False

# Все остальное - True
print(bool(1))           # True
print(bool(-1))          # True
print(bool("text"))      # True
print(bool([1]))         # True
```

## ⚠️ Частые ошибки

```python
# ❌ Забыли присвоить значение
print(undefined_var)    # NameError: name 'undefined_var' is not defined

# ✅ Присвойте сначала
value = 42
print(value)            # 42

# ❌ Неправильное имя переменной (не может начинаться с цифры)
2fast = 100             # SyntaxError
_2fast = 100            # OK
two_fast = 100          # OK

# ❌ Используете зарезервированное слово
class = 10              # SyntaxError (class - зарезервировано)
if = True               # SyntaxError (if - зарезервировано)

# ✅ Используйте другие имена
my_class = 10
my_if = True

# ❌ Путаете типы при операциях
x = "5"
y = 10
result = x + y          # TypeError: can only concatenate str with int

# ✅ Преобразуйте тип
result = int(x) + y     # 15
# или
result = x + str(y)     # "510"

# ❌ Забыли что int() округляет вниз
x = 3.7
print(int(x))           # 3 (не 4!)

# ✅ Используйте round() если нужно округлить
print(round(x))         # 4

# ❌ Бесконечная точность float (на самом деле нет)
print(0.1 + 0.2)        # 0.30000000000000004 (ошибка точности)

# ✅ Используйте round() или Decimal для точности
print(round(0.1 + 0.2, 1))  # 0.3
```
