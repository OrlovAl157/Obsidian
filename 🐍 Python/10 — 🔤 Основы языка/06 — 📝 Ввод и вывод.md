# 📝 Ввод и вывод

## 📌 Коротко

> **print()** выводит текст и значения на консоль, **input()** читает строку введённую пользователем. Это основные способы взаимодействия с пользователем в консольных приложениях.

---

## 💡 Полезные советы

- **print() выводит в консоль** — автоматически добавляет новую строку
- **input() всегда возвращает строку** — даже если введено число!
- **f-strings самые современные** — используйте их для форматирования
- **sep и end параметры print** — контролируют разделители и конец
- **Преобразуйте тип сразу** — `age = int(input(...))` вместо отдельной строки

---

## 🚀 Быстрая справка

```python
# Вывод
print("Hello")                   # Hello
print(42)                         # 42
print("A", "B", "C")             # A B C

# Форматирование
name = "Alice"
age = 25
print(f"Name: {name}, Age: {age}")  # Name: Alice, Age: 25

# Ввод
name = input("Your name: ")      # Прочитать строку
age = int(input("Your age: "))   # Прочитать и преобразовать

# Управление выводом
print("A", "B", sep="-")         # A-B
print("No newline", end="")      # Без новой строки в конце
```

## 📖 Полные примеры

### print() базовое

```python
# Простой вывод
print("Hello, World!")

# Вывод чисел
print(42)
print(3.14)
print(True)
print(None)

# Вывод нескольких значений
print("Name:", "Alice")          # Name: Alice
print(1, 2, 3)                   # 1 2 3

# Вывод переменных
x = 5
y = 10
print(x, y)                      # 5 10
```

### print() параметры

```python
# sep - разделитель между значениями (по умолчанию пробел)
print("A", "B", "C")            # A B C
print("A", "B", "C", sep="-")   # A-B-C
print("A", "B", "C", sep="")    # ABC
print("A", "B", "C", sep=" | ") # A | B | C

# end - строка в конце (по умолчанию \n - новая строка)
print("Hello", end="")          # Hello (без новой строки)
print("World")                   # World (на той же строке)
# Вывод: HelloWorld

print("Line 1", end=" -> ")
print("Line 2")
# Вывод: Line 1 -> Line 2

# file - куда выводить (по умолчанию консоль)
import sys
print("Error!", file=sys.stderr) # В stderr вместо stdout
```

### Форматирование строк

```python
# 1️⃣ f-strings (современный способ) - РЕКОМЕНДУЕТСЯ
name = "Alice"
age = 25
score = 95.5

print(f"Name: {name}")           # Name: Alice
print(f"Age: {age}")             # Age: 25
print(f"Name: {name}, Age: {age}") # Name: Alice, Age: 25

# С выражениями внутри фигурных скобок
x = 10
y = 3
print(f"{x} + {y} = {x + y}")    # 10 + 3 = 13
print(f"{x} * {y} = {x * y}")    # 10 * 3 = 30

# Форматирование чисел в f-strings
pi = 3.14159
print(f"Pi: {pi}")               # Pi: 3.14159
print(f"Pi: {pi:.2f}")           # Pi: 3.14 (2 знака после запятой)
print(f"Pi: {pi:.4f}")           # Pi: 3.1416 (4 знака)

temperature = -5
print(f"Temp: {temperature:+d}")  # Temp: -5 (с знаком)

count = 7
print(f"Count: {count:03d}")      # Count: 007 (с нулями впереди)

# 2️⃣ .format() (классический способ)
print("Name: {}, Age: {}".format(name, age))
# Name: Alice, Age: 25

print("Name: {0}, Age: {1}".format(name, age))
print("Age: {1}, Name: {0}".format(name, age))
# Age: 25, Name: Alice

print("{name} is {age} years old".format(name="Alice", age=25))
# Alice is 25 years old

# 3️⃣ % форматирование (старый способ) - не рекомендуется
print("Name: %s, Age: %d" % (name, age))
# Name: Alice, Age: 25

# 4️⃣ Конкатенация (простой способ для коротких строк)
print("Hello " + name + ", you are " + str(age) + " years old")
```

### input() базовое

```python
# Простой ввод
name = input("What is your name? ")
print(f"Hello, {name}!")

# Ввод числа (нужно преобразование)
age_str = input("How old are you? ")
age = int(age_str)
print(f"You are {age} years old")

# Более компактно
age = int(input("How old are you? "))
print(f"You are {age} years old")

# Ввод нескольких значений на одной строке
line = input("Enter x and y: ")  # Пример: "5 10"
x, y = line.split()              # Разделить по пробелам
x, y = int(x), int(y)            # Преобразовать
print(f"x={x}, y={y}")

# Или сразу с преобразованием
x, y = map(int, input("Enter x and y: ").split())
print(f"x={x}, y={y}")
```

### Обработка разных типов ввода

```python
# Строка
name = input("Name: ")
print(f"Name is: {name}")

# Целое число
age = int(input("Age: "))
print(f"Age is: {age}")

# Дробное число
height = float(input("Height: "))
print(f"Height is: {height}")

# Логическое значение (сложнее)
active_str = input("Active (yes/no): ").lower()
is_active = active_str == "yes"
print(f"Active: {is_active}")

# Список значений
numbers_str = input("Numbers (через пробел): ")
numbers = list(map(int, numbers_str.split()))
print(f"Numbers: {numbers}")
```

## 🎯 Практические примеры

### Простой калькулятор

```python
# Ввод
num1 = float(input("First number: "))
num2 = float(input("Second number: "))
operation = input("Operation (+, -, *, /): ")

# Вычисление
if operation == "+":
    result = num1 + num2
elif operation == "-":
    result = num1 - num2
elif operation == "*":
    result = num1 * num2
elif operation == "/":
    if num2 == 0:
        print("Error: Division by zero")
        exit()
    result = num1 / num2
else:
    print("Invalid operation")
    exit()

# Вывод
print(f"{num1} {operation} {num2} = {result}")
```

### Интерактивный опрос

```python
total = 0
count = 0

print("Enter numbers (enter 'done' to finish):")

while True:
    user_input = input("Number: ")
    
    if user_input.lower() == "done":
        break
    
    try:
        num = float(user_input)
        total += num
        count += 1
    except ValueError:
        print("Invalid number, try again")
        continue

if count > 0:
    average = total / count
    print(f"\nSum: {total}")
    print(f"Count: {count}")
    print(f"Average: {average:.2f}")
else:
    print("No numbers entered")
```

### Таблица вывода

```python
# Простая таблица
print("Name      Age  City")
print("-" * 30)
print("Alice     25   NYC")
print("Bob       30   LA")
print("Charlie   35   Chicago")

# С форматированием
data = [
    ("Alice", 25, "NYC"),
    ("Bob", 30, "LA"),
    ("Charlie", 35, "Chicago")
]

print(f"{'Name':<10} {'Age':<5} {'City':<10}")
print("-" * 25)
for name, age, city in data:
    print(f"{name:<10} {age:<5} {city:<10}")
# Вывод:
# Name       Age   City
# -------------------------
# Alice      25    NYC
# Bob        30    LA
# Charlie    35    Chicago
```

## ⚠️ Частые ошибки

```python
# ❌ Забыли что input возвращает строку
age = input("Age: ")        # "25" (строка!)
age += 1                    # TypeError: can only concatenate str with int

# ✅ Преобразуйте тип
age = int(input("Age: "))
age += 1

# ❌ Используете старый способ форматирования
print("Name: " + name + ", Age: " + str(age))

# ✅ Используйте f-strings
print(f"Name: {name}, Age: {age}")

# ❌ Забыли str() при конкатенации
name = "Alice"
print("Hello " + name)       # OK
print("You are " + 25 + " years old")  # TypeError

# ✅ Преобразуйте или используйте f-string
print(f"You are {25} years old")

# ❌ Не обработали исключения при преобразовании
age = int(input("Age: "))    # Если введено "abc", ValueError!

# ✅ Обработайте исключение
try:
    age = int(input("Age: "))
except ValueError:
    print("Please enter a valid number")
    age = 0

# ❌ Неправильное разделение входных данных
x, y, z = input("Enter 3 numbers: ")  # Если "1 2 3", распределит по символам

# ✅ Используйте split()
x, y, z = input("Enter 3 numbers: ").split()
x, y, z = int(x), int(y), int(z)

# или
x, y, z = map(int, input("Enter 3 numbers: ").split())
```
