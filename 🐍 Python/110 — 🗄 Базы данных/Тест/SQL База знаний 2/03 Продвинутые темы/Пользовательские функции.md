# Пользовательские функции в SQL

## Синтаксис
```sql
DELIMITER //
CREATE FUNCTION function_name(параметр1 ТИП, параметр2 ТИП, ...)
RETURNS тип_возврата
[DETERMINISTIC | NOT DETERMINISTIC]
[READS SQL DATA | NO SQL]
COMMENT 'комментарий'
BEGIN
    -- тело функции
    RETURN значение;
END //
DELIMITER ;
```

## Виды функций

| Ключевое слово | Значение |
|---|---|
| DETERMINISTIC | всегда возвращает одинаковый результат |
| NOT DETERMINISTIC | может возвращать разные результаты |
| READS SQL DATA | работает с данными из БД |
| NO SQL | не обращается к таблицам |

## Переменные внутри функции
```sql
DECLARE variable_name тип;
```

## Удаление функции
```sql
DROP FUNCTION function_name;
```

## Пример 1: чётное/нечётное
```sql
DELIMITER //
CREATE FUNCTION IS_EVEN(number INT)
RETURNS INT
DETERMINISTIC
BEGIN
    RETURN IF(number MOD 2 = 0, 1, 0);
END //
DELIMITER ;

SELECT IS_EVEN(10), IS_EVEN(15);  -- 1, 0
```

## Пример 2: с подзапросом
```sql
DELIMITER //
CREATE FUNCTION STUDENTS_WITH_GRADE(score INT)
RETURNS INT
NOT DETERMINISTIC
READS SQL DATA
BEGIN
    RETURN (SELECT COUNT(*) FROM Math WHERE grade = score);
END //
DELIMITER ;
```

## Пример 3: с переменными
```sql
DELIMITER //
CREATE FUNCTION MIN_MAX_GRADE()
RETURNS TEXT
NOT DETERMINISTIC
READS SQL DATA
BEGIN
    DECLARE min_grade INT;
    DECLARE max_grade INT;
    
    SET min_grade := (SELECT MIN(grade) FROM Math);
    SET max_grade := (SELECT MAX(grade) FROM Math);
    
    RETURN CONCAT('Lowest: ', min_grade, ', Highest: ', max_grade);
END //
DELIMITER ;
```

## Пример 4: IF-ELSEIF-ELSE
```sql
DELIMITER //
CREATE FUNCTION IS_EVEN(number INT)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE result INT;
    
    IF number MOD 2 = 0 THEN
        SET result := 1;
    ELSEIF number MOD 2 IS NULL THEN
        SET result := NULL;
    ELSE
        SET result := 0;
    END IF;
    
    RETURN result;
END //
DELIMITER ;
```

## Пример 5: WHILE
```sql
DELIMITER //
CREATE FUNCTION SUM_UP_TO_N(n INT)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE result INT DEFAULT 0;
    DECLARE count INT DEFAULT 1;
    
    WHILE count <= n DO
        SET result := result + count;
        SET count := count + 1;
    END WHILE;
    
    RETURN result;
END //
DELIMITER ;

SELECT SUM_UP_TO_N(5);  -- 15
```
