## Пример удаления информации из таблицы с счётчиком удалённых данных:

| Колонка 1 | Колонка 2 | Колонка 3 |
| --- | --- | --- |
| DELIMITER //<br>CREATE PROCEDURE DELETE_LOW_GRADE_STUDENTS()<br>BEGIN<br>    DECLARE low_grade_students INT;<br>    <br>    SET low_grade_students := (SELECT COUNT(*)<br>                               FROM Math<br>                               WHERE grade <= 2);<br>    <br>    DELETE FROM Math<br>    WHERE grade <= 2;<br>        <br>    SELECT low_grade_students AS deleted_students;<br>END //<br>DELIMITER ;<br><br>CALL DELETE_LOW_GRADE_STUDENTS; |  | +------------------------+<br>\| deleted_students  \|<br>+------------------------+<br>\| 4                           \|<br>+------------------------+ |
