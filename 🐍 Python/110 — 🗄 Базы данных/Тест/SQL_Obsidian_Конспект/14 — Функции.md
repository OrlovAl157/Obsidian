## Пример использования WHILE:

| Колонка 1 | Колонка 2 |
| --- | --- |
| DELIMITER //<br>CREATE FUNCTION SUM_UP_TO_N(n INT)<br>RETURNS INT<br>DETERMINISTIC<br>BEGIN<br>    DECLARE result INT DEFAULT 0;                -- конечная сумма<br>    DECLARE count INT DEFAULT 1;                 -- счетчик<br>   <br>    WHILE count <= n DO                          -- проверяем, что счетчик не больше n<br>        SET result := result + count;            -- прибавляем очередное натуральное число<br>        SET count := count + 1;                  -- увеличиваем счетчик на единицу<br>    END WHILE;<br>   <br>    RETURN result;<br>END //<br>DELIMITER ;<br><br>SELECT SUM_UP_TO_N(1),                          -- 1<br>       SUM_UP_TO_N(2),                          -- 1 + 2<br>       SUM_UP_TO_N(3),                          -- 1 + 2 + 3<br>       SUM_UP_TO_N(4),                          -- 1 + 2 + 3 + 4<br>       SUM_UP_TO_N(5);                          -- 1 + 2 + 3 + 4 + 5 | +--------------------------+--------------------------+---------------------------+--------------------------+---------------------------+<br>\| SUM_UP_TO_N(1) \| SUM_UP_TO_N(2) \| SUM_UP_TO_N(3) \| SUM_UP_TO_N(4) \| SUM_UP_TO_N(5) \|<br>+---------------------------+--------------------------+---------------------------+--------------------------+--------------------------+<br>\| 1                               \| 3                              \| 6                              \| 10                           \| 15                            \|<br>+---------------------------+--------------------------+---------------------------+--------------------------+--------------------------+ |
