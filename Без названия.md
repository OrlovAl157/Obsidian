`import logging

# ШАГ 1: Создаю логгер
logger = logging.getLogger('myapp')
logger.setLevel(logging.DEBUG)  # Логгер пропускает сообщения от DEBUG и выше

# ШАГ 2: Создаю handler (куда отправлять)
console_handler = logging.StreamHandler()  # В консоль
file_handler = logging.FileHandler('app.log')  # В файл

# ШАГ 3: Настраиваю каждый handler
console_handler.setLevel(logging.WARNING)  # Консоль видит только WARNING и выше
file_handler.setLevel(logging.DEBUG)  # Файл видит всё от DEBUG

# ШАГ 4: Создаю форматтер (как выглядит сообщение)
formatter = logging.Formatter(
    '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)

# ШАГ 5: Применяю форматтер к handlers
console_handler.setFormatter(formatter)
file_handler.setFormatter(formatter)

# ШАГ 6: Добавляю handlers к логгеру
logger.addHandler(console_handler)
logger.addHandler(file_handler)

# РЕЗУЛЬТАТ:
logger.debug("Отладочное сообщение")  # → только в файл
logger.warning("Ошибка!")  # → в файл И консоль
logger.critical("Критично!")  # → в файл И консоль
``
